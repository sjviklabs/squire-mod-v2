**SquireItemHandler.java** (`src/main/java/com/sjviklabs/squire/inventory/SquireItemHandler.java`):

Extends ItemStackHandler (which implements IItemHandler). Total slots = 6 equipment + tier.getBackpackSlots().

Equipment slots are indices 0-5:
  - Slot 0: Helmet
  - Slot 1: Chestplate
  - Slot 2: Leggings
  - Slot 3: Boots
  - Slot 4: Main hand weapon
  - Slot 5: Off hand (shield)

Backpack slots start at index 6. getSlots() returns 6 + squire.getTier().getBackpackSlots() dynamically.

```java
package com.sjviklabs.squire.inventory;

import com.sjviklabs.squire.entity.SquireEntity;
import net.neoforged.neoforge.items.ItemStackHandler;
import net.minecraft.world.item.ItemStack;
import org.jetbrains.annotations.NotNull;

public class SquireItemHandler extends ItemStackHandler {

    public static final int EQUIPMENT_SLOTS = 6;
    public static final int SLOT_HELMET = 0;
    public static final int SLOT_CHEST = 1;
    public static final int SLOT_LEGS = 2;
    public static final int SLOT_BOOTS = 3;
    public static final int SLOT_MAINHAND = 4;
    public static final int SLOT_OFFHAND = 5;

    private final SquireEntity squire;

    public SquireItemHandler(SquireEntity squire) {
        // Initialize with champion max to avoid resize issues; getSlots() gates access
        super(EQUIPMENT_SLOTS + 36); // 42 max slots
        this.squire = squire;
    }

    @Override
    public int getSlots() {
        return EQUIPMENT_SLOTS + squire.getTier().getBackpackSlots();
    }

    /** True if this slot index is an equipment slot (0-5). */
    public boolean isEquipmentSlot(int slot) {
        return slot < EQUIPMENT_SLOTS;
    }

    /** True if this slot index is a backpack slot (6+) and within current tier capacity. */
    public boolean isBackpackSlot(int slot) {
        return slot >= EQUIPMENT_SLOTS && slot < getSlots();
    }

    @Override
    public @NotNull ItemStack insertItem(int slot, @NotNull ItemStack stack, boolean simulate) {
        // Reject inserts beyond current tier capacity
        if (slot >= getSlots()) return stack;
        return super.insertItem(slot, stack, simulate);
    }

    @Override
    public @NotNull ItemStack extractItem(int slot, int amount, boolean simulate) {
        if (slot >= getSlots()) return ItemStack.EMPTY;
        return super.extractItem(slot, amount, simulate);
    }

    @Override
    protected void onContentsChanged(int slot) {
        squire.setChanged(); // Marks entity dirty so NBT is saved
    }
}
```

```java
package com.sjviklabs.squire.inventory;

import com.sjviklabs.squire.SquireRegistry;
import com.sjviklabs.squire.entity.SquireEntity;
import net.minecraft.world.entity.player.Inventory;
import net.minecraft.world.entity.player.Player;
import net.minecraft.world.inventory.AbstractContainerMenu;
import net.minecraft.world.inventory.Slot;
import net.minecraft.world.item.ItemStack;
import net.neoforged.neoforge.items.SlotItemHandler;

public class SquireMenu extends AbstractContainerMenu {

    private final SquireEntity squire;
    private final SquireItemHandler itemHandler;

    /** Server-side constructor (called from SquireEntity.mobInteract) */
    public SquireMenu(int windowId, Inventory playerInventory, SquireEntity squire) {
        super(SquireRegistry.SQUIRE_MENU.get(), windowId);
        this.squire = squire;
        this.itemHandler = squire.getItemHandler();

        // Add squire backpack slots (slots 6+ in handler = backpack)
        // Phase 5 will arrange these in the GUI grid; here they just need to be present
        int backpackSlots = squire.getTier().getBackpackSlots();
        for (int i = 0; i < backpackSlots; i++) {
            int handlerSlot = SquireItemHandler.EQUIPMENT_SLOTS + i;
            this.addSlot(new SlotItemHandler(this.itemHandler, handlerSlot,
                8 + (i % 9) * 18,         // x — row-major, 9 per row
                18 + (i / 9) * 18));       // y — stacked rows
        }

        // Add player inventory slots (27 main + 9 hotbar = standard layout)
        for (int row = 0; row < 3; row++) {
            for (int col = 0; col < 9; col++) {
                this.addSlot(new Slot(playerInventory, col + row * 9 + 9,
                    8 + col * 18, 100 + row * 18));
            }
        }
        for (int col = 0; col < 9; col++) {
            this.addSlot(new Slot(playerInventory, col, 8 + col * 18, 158));
        }
    }

    @Override
    public ItemStack quickMoveStack(Player player, int index) {
        // Standard shift-click implementation — move between squire and player inventory
        ItemStack result = ItemStack.EMPTY;
        Slot slot = this.slots.get(index);
        if (slot.hasItem()) {
            ItemStack slotStack = slot.getItem();
            result = slotStack.copy();
            int squireSlots = squire.getTier().getBackpackSlots();
            if (index < squireSlots) {
                if (!this.moveItemStackTo(slotStack, squireSlots, this.slots.size(), true)) {
                    return ItemStack.EMPTY;
                }
            } else {
                if (!this.moveItemStackTo(slotStack, 0, squireSlots, false)) {
                    return ItemStack.EMPTY;
                }
            }
            if (slotStack.isEmpty()) slot.set(ItemStack.EMPTY);
            else slot.setChanged();
        }
        return result;
    }

    @Override
    public boolean stillValid(Player player) {
        // Menu stays open as long as squire is alive and within 8 blocks
        return squire.isAlive() && squire.distanceTo(player) < 8.0;
    }
}
```
