```java
@Test
void codecRoundTrip() {
    SquireDataAttachment.SquireData original = SquireDataAttachment.SquireData.empty()
        .withXP(1500, 12)
        .withName("Aldric")
        .withAppearance(true);
    JsonElement encoded = SquireDataAttachment.SquireData.CODEC
        .encodeStart(JsonOps.INSTANCE, original)
        .getOrThrow();
    SquireDataAttachment.SquireData decoded = SquireDataAttachment.SquireData.CODEC
        .parse(JsonOps.INSTANCE, encoded)
        .getOrThrow();
    assertEquals(original.totalXP(), decoded.totalXP());
    assertEquals(original.level(), decoded.level());
    assertEquals(original.customName(), decoded.customName());
    assertEquals(original.slimModel(), decoded.slimModel());
}
```
