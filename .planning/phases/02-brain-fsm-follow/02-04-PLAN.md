    Fields:
      private final SquireEntity squire
      private int eatCooldown = 0
      private static final int EAT_DURATION = 40  // 2 seconds eating animation

    shouldEat():
      float ratio = squire.getHealth() / squire.getMaxHealth()
      return ratio < SquireConfig.eatHealthThreshold.get() && eatCooldown <= 0

    startEating():
      // Find food in inventory using IItemHandler
      IItemHandler inv = getInventory()
      if (inv == null) return false
      for (int i = 0; i < inv.getSlots(); i++) {
          ItemStack stack = inv.getStackInSlot(i)  // READ ONLY — do not mutate
          if (stack.getItem() instanceof net.minecraft.world.item.food.FoodProperties food
              || stack.getFoodProperties(squire) != null) {
              // Found food — consume via extractItem, not shrink()
              ItemStack consumed = inv.extractItem(i, 1, false)
              if (!consumed.isEmpty()) {
                  float healAmount = 2.0f  // conservative; full food properties complex in 1.21.1
                  squire.heal(healAmount)
                  eatCooldown = 200  // 10-second cooldown before eating again
                  return true
              }
          }
      }
      return false

    Note on food healing in 1.21.1: FoodProperties API changed in 1.21. Use
    stack.get(net.minecraft.core.component.DataComponents.FOOD) to get FoodComponent.
    If FoodComponent is present, call component.nutrition() for heal amount.
    Fallback: if API not available at compile time, use flat 2.0f heal.

    reset():
      eatCooldown = 0

    tick():
      if (eatCooldown > 0) eatCooldown--
