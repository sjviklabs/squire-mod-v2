**Placeholder PNG textures** — use Python to generate minimal 64x64 solid-color PNG files. Each texture needs to be a valid PNG; solid color placeholders are fine.

```python
import struct, zlib

def make_placeholder_png(path, r, g, b):
    import os
    os.makedirs(os.path.dirname(path), exist_ok=True)
    w, h = 64, 64
    # Raw image data: RGBA rows
    raw = b''
    for _ in range(h):
        raw += b'\x00' + bytes([r, g, b, 255]) * w
    compressed = zlib.compress(raw)
    def chunk(name, data):
        c = name + data
        return struct.pack('>I', len(data)) + c + struct.pack('>I', zlib.crc32(c) & 0xffffffff)
    png = (b'\x89PNG\r\n\x1a\n'
           + chunk(b'IHDR', struct.pack('>IIBBBBB', w, h, 8, 2, 0, 0, 0))
           + chunk(b'IDAT', compressed)
           + chunk(b'IEND', b''))
    with open(path, 'wb') as f:
        f.write(png)
```
