```java
package com.sjviklabs.squire.item;

import net.minecraft.world.entity.LivingEntity;
import net.minecraft.world.item.SwordItem;
import net.minecraft.world.item.Tiers;
import net.minecraft.world.item.Item;
import net.minecraft.world.level.Level;
import net.minecraft.world.phys.AABB;
import java.util.List;

public class SquireHalberdItem extends SwordItem {

    // SwordItem(Tier, Properties) — Tiers.NETHERITE base, then we set ATK via attribute modifiers
    public SquireHalberdItem() {
        super(Tiers.NETHERITE, new Item.Properties()
            .attributes(SwordItem.createAttributes(Tiers.NETHERITE, 2, -3.0F)) // 2 + Netherite(6) = 8 base; override to 7 via net modifier
        );
        // NeoForge 21.1: use SwordItem.createAttributes(tier, attackDamageBonus, attackSpeedBonus)
        // attackDamageBonus=2 + Netherite base(6) - 1 (hand) = 7 effective. attackSpeedBonus=-3.0F
        // Reach +1.0: add via NeoForge entity_reach_distance attribute in createAttributes or override getAttributeModifiers
    }

    /**
     * Performs 360-degree sweep AoE around the attacker.
     * Called by CombatHandler when halberdHitCount reaches threshold.
     * Predicate excludes attacker, squire's owner, and other players.
     */
    public static void performSweep(LivingEntity attacker, LivingEntity primaryTarget, float damage) {
        Level level = attacker.level();
        if (level.isClientSide()) return;
        
        AABB sweepArea = attacker.getBoundingBox().inflate(2.5);
        List<LivingEntity> targets = level.getEntitiesOfClass(LivingEntity.class, sweepArea,
            e -> e != attacker
              && e != primaryTarget
              && !e.getType().is(net.minecraft.tags.EntityTypeTags.FALL_DAMAGE_IMMUNE) // proxy for owner-like
              && !(e instanceof net.minecraft.world.entity.player.Player));
        
        for (LivingEntity target : targets) {
            target.hurt(level.damageSources().mobAttack(attacker), damage);
        }
    }
}
```
