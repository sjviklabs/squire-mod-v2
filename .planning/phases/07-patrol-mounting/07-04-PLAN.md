```java
// tickMountedCombat() pattern:
public SquireAIState tickMountedCombat(SquireEntity s) {
    if (!isMounted()) return SquireAIState.IDLE;
    AbstractHorse horse = (AbstractHorse) s.getVehicle();

    // 1. Check horse HP — dismount if below threshold
    float hpFraction = horse.getHealth() / horse.getMaxHealth();
    if (hpFraction < (float) SquireConfig.horseFleeThreshold.get()) {
        orderDismount();
        return SquireAIState.IDLE;
    }

    // 2. Get combat target (from CombatHandler or SquireEntity.getTarget())
    LivingEntity target = s.getTarget();
    if (target == null || !target.isAlive()) {
        return SquireAIState.MOUNTED_FOLLOW; // no target — return to follow
    }

    // 3. Drive horse toward target using existing driveHorseToward()
    driveHorseToward(s, horse, target.getX(), target.getY(), target.getZ(), 1.0F);

    // 4. Delegate melee attack to CombatHandler (Phase 4) or direct doHurtTarget()
    // Reach: meleeRange + 1.5 to account for mounted height
    double mountedReach = SquireConfig.meleeRange.get() + 1.5;
    if (s.distanceToSqr(target) <= mountedReach * mountedReach) {
        if (brain.getCombatHandler() != null) {
            brain.getCombatHandler().performMeleeAttack(s, target);
        } else {
            s.doHurtTarget(target); // fallback if CombatHandler absent
        }
    }

    // 5. Block ranged combat while mounted — no bow use
    return SquireAIState.MOUNTED_COMBAT;
}
```
